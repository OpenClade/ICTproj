FROM Student

ON Student.student_id = Attendance.student_id
WHERE (Attendance.attendance_num < 70 OR Student.student_gpa <= 2) AND Student.major_id = 1005 AND Student.student_gender = 'Female'
ORDER BY Attendance.attendance_num
